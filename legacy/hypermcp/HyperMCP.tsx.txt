// Status indicator component
const StatusBadge = ({ status }: { status: string }) => {
  const statusStyles = {
    ok: 'bg-emerald-900 text-emerald-100 border-emerald-700',
    idle: 'bg-gray-800 text-gray-300 border-gray-700',
    degraded: 'bg-amber-900 text-amber-100 border-amber-700',
    error: 'bg-red-900 text-red-100 border-red-700',
    disconnected: 'bg-gray-900 text-gray-400 border-gray-800',
  };

  const dotColors = {
    ok: '#22C98A',
    idle: '#6B7280',
    degraded: '#FBBF24',
    error: '#F87171',
    disconnected: '#4B5563',
  };

  return (
    <div className="flex items-center gap-2">
      <div
        className="w-1.5 h-1.5 rounded-full"
        style={{ backgroundColor: dotColors[status as keyof typeof dotColors] }}
      />
      <span className={`text-xs font-mono px-2 py-1 rounded border ${statusStyles[status as keyof typeof statusStyles]}`}>
        {status === 'ok' ? '●' : status === 'idle' ? '◐' : '◉'} {status.toUpperCase()}
      </span>
    </div>
  );
};
